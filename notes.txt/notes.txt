        """

        if not os.path.exists(TEMP_DIR):

        # Unique folder for each repo
        repo_id = str(uuid.uuid4())
        repo_path = os.path.join(TEMP_DIR, repo_id)

        try:
            subprocess.run(

github_service.py
    import os
    import subprocess
    import shutil
    import uuid
    import re

    TEMP_DIR = "temp_repos"


    # -------------------------------
    # Validate GitHub URL
    # -------------------------------
    def validate_github_url(repo_url: str):
        return repo_url is not None and "github.com" in repo_url


    # -------------------------------
    # Extract repo name
    # -------------------------------
    def extract_repo_name(repo_url: str):
        """
        Example:
        https://github.com/facebook/react -> react
        """
        match = re.search(r"github\.com/.+?/(.+)$", repo_url)

        if not match:
            return None

        return match.group(1)


    # -------------------------------
    # Clone repository
    # -------------------------------
    def clone_github_repo(repo_url: str):
        """
        Clone GitHub repo into temp folder
        """

        if not os.path.exists(TEMP_DIR):
            os.makedirs(TEMP_DIR)

        repo_id = str(uuid.uuid4())
        repo_path = os.path.join(TEMP_DIR, repo_id)

        try:
            subprocess.run(
                ["git", "clone", "--depth", "1", repo_url, repo_path],
                check=True
            )

            return repo_path

        except Exception as e:
            print("❌ Git clone failed:", e)

            # cleanup partial folder
            if os.path.exists(repo_path):
                shutil.rmtree(repo_path)

            return None


    # -------------------------------
    # Get repo file structure
    # -------------------------------
    def get_repo_tree(repo_path: str):
        """
        Returns structured file list with extensions
        """

        tree = []

        for root, dirs, files in os.walk(repo_path):

            # skip git internals
            if ".git" in root:
                continue

            for file in files:
                full_path = os.path.join(root, file)
                relative_path = os.path.relpath(full_path, repo_path)

                tree.append({
                    "path": relative_path,
                    "extension": os.path.splitext(file)[1]
                })

        return tree


    # -------------------------------
    # Cleanup repo
    # -------------------------------
    def cleanup_repo(repo_path: str):
        """
        Delete cloned repo after processing
        """

        try:
            shutil.rmtree(repo_path)
        except Exception as e:
            print("⚠️ Cleanup failed:", e)

    from fastapi import APIRouter
    from pydantic import BaseModel

    from app.services.github_service import (
        validate_github_url,
        extract_repo_name,
        clone_github_repo,
        get_repo_tree,
        cleanup_repo
    )

    router = APIRouter()


    class RepoRequest(BaseModel):
        repo_url: str


    @router.post("/analyze")
    def analyze_repo(data: RepoRequest):

        repo_url = data.repo_url

        print(f"📥 Repository URL received: {repo_url}")

        # Validate GitHub URL
        if not validate_github_url(repo_url):
            return {
                "success": False,
                "message": "Invalid GitHub URL"
            }

        # Extract repo name
        repo_name = extract_repo_name(repo_url)

        # Clone repo
        repo_path = clone_github_repo(repo_url)

        if not repo_path:
            return {
                "success": False,
                "message": "Failed to clone repository"
            }

        # Get file structure
        repo_tree = get_repo_tree(repo_path)

        # Cleanup
        cleanup_repo(repo_path)

        # Build response
        return {
            "success": True,
            "repo_name": repo_name,
            "total_files": len(repo_tree),
            "files": repo_tree[:25],  # limit output
            "summary": {
                "python_files": len([f for f in repo_tree if f["extension"] == ".py"]),
                "js_files": len([f for f in repo_tree if f["extension"] in [".js", ".jsx", ".ts", ".tsx"]]),
                "json_files": len([f for f in repo_tree if f["extension"] == ".json"])
            }
        }

hero.txt
    import axios from "axios";
import { useState } from "react";

type FileItem = {
  path: string;
  extension: string;
};

export default function Hero() {
  const [input, setInput] = useState<string>("");
  const [repoName, setRepoName] = useState<string>("");
  const [totalFiles, setTotalFiles] = useState<number>(0);
  const [files, setFiles] = useState<FileItem[]>([]);
  const [loading, setLoading] = useState<boolean>(false);
  const [error, setError] = useState<string>("");

  const handleAnalyze = async () => {
    if (!input.trim()) {
      setError("Please enter a GitHub repository URL.");
      return;
    }

    try {
      setLoading(true);
      setError("");

      const res = await axios.post("http://127.0.0.1:8000/analyze", {
        repo_url: input.trim(),
      });

      if (!res.data || !res.data.success) {
        setError(res.data?.message || "Backend returned an error.");
        return;
      }

      setRepoName(res.data.repo_name || "Repository analyzed");
      setTotalFiles(res.data.total_files || 0);
      setFiles(res.data.files || []);
    } catch (err: any) {
      console.log("API ERROR:", err);
      setError(err?.response?.data?.message || "Backend request failed.");
    } finally {
      setLoading(false);
    }
  };

  return (
    <main className="w-full max-w-5xl px-6 py-16">
      <section className="text-center">
        <div className="mb-4 inline-flex rounded-full bg-blue-100 px-4 py-2 text-sm font-medium text-blue-700">
          Developer Onboarding Assistant
        </div>

        <h1 className="text-5xl font-bold tracking-tight text-gray-900 md:text-7xl">
          RepoLens
        </h1>

        <p className="mx-auto mt-5 max-w-2xl text-lg text-gray-600">
          Paste a GitHub repository link and understand its structure, files,
          and tech stack faster.
        </p>
      </section>

      <section className="mx-auto mt-10 max-w-3xl rounded-2xl border border-gray-200 bg-white p-5 shadow-xl">
        <div className="flex flex-col gap-3 md:flex-row">
          <input
            value={input}
            onChange={(e) => setInput(e.target.value)}
            placeholder="https://github.com/owner/repository"
            className="flex-1 rounded-xl border border-gray-300 px-4 py-3 text-gray-800 outline-none transition focus:border-blue-500 focus:ring-4 focus:ring-blue-100"
          />

          <button
            onClick={handleAnalyze}
            disabled={loading}
            className="rounded-xl bg-blue-600 px-6 py-3 font-semibold text-white transition hover:bg-blue-700 disabled:cursor-not-allowed disabled:bg-blue-300"
          >
            {loading ? "Analyzing..." : "Analyze"}
          </button>
        </div>

        {error && (
          <div className="mt-4 rounded-xl border border-red-200 bg-red-50 px-4 py-3 text-sm text-red-700">
            {error}
          </div>
        )}
      </section>

      {repoName && (
        <section className="mx-auto mt-8 max-w-3xl rounded-2xl border border-gray-200 bg-white p-6 shadow-lg">
          <div className="flex items-start justify-between gap-4">
            <div>
              <p className="text-sm font-medium text-gray-500">
                Analysis Result
              </p>
              <h2 className="mt-1 text-2xl font-bold text-gray-900">
                {repoName}
              </h2>
            </div>

            <div className="rounded-xl bg-gray-100 px-4 py-3 text-center">
              <p className="text-2xl font-bold text-gray-900">{totalFiles}</p>
              <p className="text-xs text-gray-500">Files</p>
            </div>
          </div>

          <div className="mt-6">
            <h3 className="mb-3 font-semibold text-gray-800">
              Repository Files
            </h3>

            {files.length === 0 ? (
              <p className="rounded-xl bg-gray-50 p-4 text-sm text-gray-500">
                No file list received yet.
              </p>
            ) : (
              <div className="max-h-80 overflow-y-auto rounded-xl border border-gray-200">
                {files.map((f, i) => (
                  <div
                    key={i}
                    className="border-b border-gray-100 px-4 py-3 text-sm text-gray-700 last:border-b-0"
                  >
                    📄 {f.path}
                  </div>
                ))}
              </div>
            )}
          </div>
        </section>
      )}
    </main>
  );
}
